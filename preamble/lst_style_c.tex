    keywordstyle = [1]\color{vsKeyword},

    emph      = [1]{if, else, for, while, break, continue, return},
    emphstyle = [2]\color{vsFunction},
    emph      = [2]{%
        fopen, fclose, fread, fscanf, pow, ceil, isalpha,

        readCipher, getAverageIOCs, setFrequency, getIOC, 
        getBestKeySize, setEngFrequency, setCaesarShift, 
        getChiSq, getChiMin, getDecrypt
        },
    emphstyle = [3]\color{vsType},
    emph      = [3]{FILE, uint8_t, uint32_t, uint64_t},
    literate = *%
        {0}{{{\color{vscNumber}0}}}{1}%
        {1}{{{\color{vscNumber}1}}}{1}%
        {2}{{{\color{vscNumber}2}}}{1}%
        {3}{{{\color{vscNumber}3}}}{1}%

        {9}{{{\color{vscNumber}9}}}{1}%
        % {0x}{{{\color{vscNumber}0x}}}{2}%
        % {0X}{{{\color{vscNumber}0X}}}{2}%
}
